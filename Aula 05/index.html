        import pandas as pd
        import numpy as np
        
        def remover_outliers(df):
            """
            Função para remover outliers de um DataFrame baseado no método de IQR (Intervalo Interquartil).
            Args:
            df (pd.DataFrame): O DataFrame de entrada com os dados.
        
            Returns:
            pd.DataFrame: O DataFrame sem os outliers.
            """
            # Criação de um novo DataFrame para não modificar o original
            df_sem_outliers = df.copy()
            
            # Para cada coluna numérica, calcular o IQR e remover outliers
            for coluna in df.select_dtypes(include=[np.number]).columns:
                # Calcular Q1 (25º percentil) e Q3 (75º percentil)
                Q1 = df[coluna].quantile(0.25)
                Q3 = df[coluna].quantile(0.75)
                
                # Calcular o IQR (Intervalo Interquartil)
                IQR = Q3 - Q1
                
                # Definir limites inferior e superior
                limite_inferior = Q1 - 1.5 * IQR
                limite_superior = Q3 + 1.5 * IQR
                
                # Filtrar os valores que estão dentro dos limites
                df_sem_outliers = df_sem_outliers[(df_sem_outliers[coluna] >= limite_inferior) & 
                                                (df_sem_outliers[coluna] <= limite_superior)]
            
            return df_sem_outliers
        
        # Exemplo de uso com um DataFrame fictício
        if __name__ == "__main__":
            # Criar um DataFrame de exemplo com valores artificiais
            data = {
                'coluna1': [1, 2, 3, 4, 5, 100, 6, 7, 8, 9],
                'coluna2': [10, 15, 14, 13, 1000, 15, 14, 15, 16, 17],
                'coluna3': [100, 200, 150, 300, 400, 5000, 600, 700, 800, 900]
            }
            df = pd.DataFrame(data)
        
            print("DataFrame original:")
            print(df)
        
            # Remover outliers
            df_sem_outliers = remover_outliers(df)
        
            print("\nDataFrame após remoção dos outliers:")
            print(df_sem_outliers)
